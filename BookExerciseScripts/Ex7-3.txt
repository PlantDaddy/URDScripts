REM TITLE Exercise 7-3
REM AUTHOR Crash
REM_BLOCK DESCRIPTION
    Modify exercise 6-2 to ensure caps lock is always on
END_REM

ATTACKMODE HID STORAGE VID_05AC PID_021E MAN_FUCK PROD_ICE SERIAL_111111
DELAY 3000

COMMAND SPACE
DELAY 250
STRINGLN textedit
DELAY 750
STRINGLN Press USB Rubber Ducky button, or not. Within the next 5 seconds
DELAY 5000
IF $_BUTTON_PUSH_RECEIVED THEN
    LED_G
    DELAY 5000
    STRINGLN Success!
ELSE IF ($_BUTTON_PUSH_RECEIVED == FALSE) THEN
    LED_R
    DELAY 5000
    STRINGLN Fail!
END_IF
WHILE TRUE
    WAIT_FOR_CAPS_OFF
    CAPSLOCK
END_WHILE
