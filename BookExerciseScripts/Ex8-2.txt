   MultiOS: Write a function with a return value, execute it, and compare it

ATTACKMODE HID STORAGE VID_05AC PID_021E MAN_FUCK PROD_ICE SERIAL_111111
FUNCTION CAPSLOCKED ()
    IF $_CAPSLOCK_ON THEN
        RETURN TRUE
    ELSE IF ( $_CAPSLOCK_ON == FALSE ) THEN
        RETURN FALSE
    END_IF
END_FUNCTION
DELAY 3000

IF CAPSLOCKED() THEN
    STRINGLN Caps is on!
END_IF
