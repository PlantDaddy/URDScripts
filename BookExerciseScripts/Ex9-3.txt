REM TITLE Exercise 9-3
REM AUTHOR Crash
REM_BLOCK DESCRIPTION
   Mac: Blink the LED red or green randomly for 10 seconds, while also randomly
    blinking the different lock keys (scroll, caps, num)
END_REM

ATTACKMODE HID STORAGE VID_05AC PID_021E MAN_FUCK PROD_ICE SERIAL_111111
$_RANDOM_MIN = 1
VAR $RAN_LOCK = 0

FUNCTION BLINK_LED ()
    $_RANDOM_MAX = 2
    VAR $RAN_LED = $_RANDOM_INT
    LED_OFF
    IF ( $RAN_LED == 1 ) THEN
        LED_G
    ELSE IF ( $RAN_LED == 2 ) THEN
        LED_R
    END_IF
    $RAN_LED = $_RANDOM_INT
END_FUNCTION

FUNCTION BLINK_LOCKS ()
    $_RANDOM_MAX = 3
    $RAN_LOCK = $_RANDOM_INT
    IF $_SCROLLLOCK_ON THEN
        SCROLLLOCK
    END_IF
    IF $_NUMLOCK_ON THEN
        NUMLOCK
    END_IF
    IF $_CAPSLOCK_ON THEN
        CAPSLOCK
    END_IF
    IF ( $RAN_LOCK == 1 ) THEN
        SCROLLLOCK
        SCROLLLOCK
    ELSE IF ( $RAN_LOCK == 2 ) THEN
        NUMLOCK
        NUMLOCK
    ELSE IF ( $RAN_LOCK == 3 ) THEN
        CAPSLOCK
        CAPSLOCK
    END_IF
END_FUNCTION
DELAY 3000

VAR $TIMER = 40
WHILE ( $TIMER > 0 )
    BLINK_LOCKS()
    BLINK_LED()
    DELAY 250
END_WHILE
