REM TITLE Exercise 10-1
REM AUTHOR Crash
REM_BLOCK DESCRIPTION
   MultiOS: Hold the shift key as long as caps lock is on
END_REM

ATTACKMODE HID STORAGE VID_RANDOM PID_RANDOM MAN_RANDOM PROD_RANDOM SERIAL_RANDOM
DELAY 3000

WHILE TRUE
    IF $_CAPSLOCK_ON THEN
        INJECT_MOD
        HOLD SHIFT
    ELSE IF ( $_CAPSLOCK_ON ) THEN
        INJECT_MOD
        RELEASE SHIFT
    END_IF
END_WHILE
