REM TITLE Exercise 9-1
REM AUTHOR Crash
REM_BLOCK DESCRIPTION
   Mac: Open a terminal, create 20 random files on the logged in user's desktop
    with random names
END_REM

ATTACKMODE HID STORAGE VID_05AC PID_021E MAN_FUCK PROD_ICE SERIAL_111111
VAR $CHARS = 0
VAR $FILES = 0
FUNCTION RAND_CHARS ()
    WHILE ( $CHARS < 20 )
        RANDOM_CHAR
        $CHARS = $CHARS + 1
    END_WHILE
    $CHARS = 0
END_FUNCTION
DELAY 3000

COMMAND SPACE
DELAY 250
STRINGLN terminal
DELAY 750
STRINGLN cd ~
DELAY 250
WHILE ( $FILES < 20 )
    STRING echo "
    RAND_CHARS()
    STRING " &> 
    RAND_CHARS()
    STRINGLN .txt
    DELAY 250
    $FILES = $FILES + 1
END_WHILE
